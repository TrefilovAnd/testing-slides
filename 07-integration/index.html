Решение


// crawline.js

function crawline(text, cb) {
    const letters = text.split('');

    function print() {
        if (!letters.length) return cb();

        process.stdout.write(letters.shift());
        setTimeout(print, 100);
    }
    print();
}

Faking time


beforeEach(() => clock = sinon.useFakeTimers());
afterEach(() => {
    clock.restore();
    process.stdout.write.restore();
});
        

it('shoult ptint text', done => {
    const write = sinon.spy(process.stdout, 'write');

    crawline('I don’t always bend time and ' +
        'space in unit tests, but when I do, ' +
        'I use Buster.JS + Sinon.JS', () => {
            assert.equal(write.callCount, 91);
            done();
        });

    clock.tick(10000);
});

Конвертор Условие

Написать универсальный конвертор величин.

Конвертор Примеры

  • 1000mm -> 1m
  • 1m -> 3.281ft
  • 1.5week -> 15120min

    .to('min'); // 15120

const express = require('express');
const convert = require('convert-units')
const app = express();

app.get('/convert', (req, res) => {
    const result = convert(req.query.value)
        .from(req.query.from)
        .to(req.query.to);
    res.json({ result });
});

app.listen(3000);

Тестирование в браузере

Web-конвертор Условие

Реализовать веб-версию универсального конвертора величин.

Web-конвертор Решение


// app.js
const express = require('express');
const convert = require('convert-units');
const app = express();

app.use(express.static('public'));
app.get('/convert', (req, res) => {
    const result = convert(req.query.value)
        .from(req.query.from)
        .to(req.query.to);
    res.json({ result });
});

app.listen(3000);
