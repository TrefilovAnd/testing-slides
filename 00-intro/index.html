Например


function getPlural(count, forms) {
    if (count > 10 && count < 15)
        return forms[2];

    var residue = count % 10;

    if (residue === 1)
        return forms[0];
    if (residue > 0 && residue < 5)
        return forms[1];
    return forms[2];
}
        

getPlural(1, ['монета', 'монеты', 'монет']); // монета
getPlural(15, ['монета', 'монеты', 'монет']); // монет
